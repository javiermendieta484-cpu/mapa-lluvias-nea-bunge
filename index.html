import pandas as pd import folium from folium.features import DivIcon from folium.plugins import FloatImage # Cargar datos df = pd.read_csv('lluvias.csv') print(f"Se cargaron {len(df)} localidades") # Crear mapa con estilo más bonito m = folium.Map( location=[-28.0, -61.5], zoom_start=7.8, tiles="https://mt1.google.com/vt/lyrs=m&x={x}&y={y}&z={z}", # ← Estilo Google Maps attr="Google Maps", name="Google Maps Style" ) # Función para elegir color según cantidad de lluvia def get_color(lluvia): if lluvia >= 100: return "#d73027" # Rojo - mucha lluvia elif lluvia >= 50: return "#fc8d59" # Naranja elif lluvia >= 25: return "#fee08b" # Amarillo elif lluvia >= 10: return "#91cf60" # Verde claro else: return "#1a9850" # Verde oscuro - poca lluvia # Agregar círculos y etiquetas for idx, row in df.iterrows(): color = get_color(row['lluvia_mm']) # Círculo con color según lluvia folium.CircleMarker( location=[row['lat'], row['lon']], radius=9, popup=f"{row['localidad']}
{row['lluvia_mm']} mm", color="black", weight=1.5, fill=True, fill_color=color, fill_opacity=0.85 ).add_to(m) # Etiqueta más limpia folium.Marker( location=[row['lat'], row['lon']], icon=DivIcon( icon_size=(80, 25), icon_anchor=(5, 18), html=f'''
{row['lluvia_mm']}
{row['localidad']}
''' ) ).add_to(m) # Agregar leyenda legend_html = '''
